电子税务局目录结构 
├─站点
  │─layui//layui框架库电子税务局专用
  │  ├─css //css目录
  │  │ │─modules //模块css目录（一般如果模块相对较大，我们会单独提取，比如下面三个：）
  │  │ │  ├─laydate
  │  │ │  ├─layer
  │  │ │  └─layim
  │  │ └─layui.css //核心样式文件
  │  ├─font  //字体图标目录
  │  ├─images //图片资源目录（目前只有layim和编辑器用到的GIF表情）
  │  │─lay //模块核心目录
  │  │─layui.js //基础核心库
  │  └─layui.all.js //包含layui.js和所有模块的合并文件  
  │─tax-css//电子税务局专用样式
  │    └─ common.css
  │─tax-font-icon//字体图标库
  │    └─ iconfont.css
  │─tax-images//电子税务局专用图库
  │─tax-module//电子税务局非layui内置组件
  │─tax-js//电子税务局专用脚本库
  │─tax-module//独立模块
